### src/App.jsx
jsx


import React, { useState, useEffect } from 'react'
import Header from './components/Header'
import StatsCard from './components/StatsCard'
import AgentStatus from './components/AgentStatus'

const agents = [
  { key: 'infra_sysadmin', name: 'مدير النظام', layer: 'infra', status: 'active' },
  { key: 'infra_monitor', name: 'وكيل المراقبة', layer: 'infra', status: 'active' },
  { key: 'infra_bugfixer', name: 'المصلح الجراحي', layer: 'infra', status: 'active' },
  { key: 'infra_builder', name: 'وكيل التطوير', layer: 'infra', status: 'active' },
  { key: 'codegen', name: 'مولّد الأكواد', layer: 'service', status: 'active' },
  { key: 'reviewer', name: 'مراجع الأكواد', layer: 'service', status: 'active' },
  { key: 'fixer', name: 'مصلح الأكواد', layer: 'service', status: 'active' },
  { key: 'planner', name: 'وكيل التخطيط', layer: 'service', status: 'active' },
  { key: 'strategic', name: 'الوكيل الاستراتيجي', layer: 'service', status: 'active' },
  { key: 'filemanager', name: 'مدير الملفات', layer: 'service', status: 'active' },
  { key: 'translator', name: 'وكيل الترجمة', layer: 'service', status: 'active' },
  { key: 'seo', name: 'محلل السيو', layer: 'service', status: 'active' },
]

export default function App() {
  const [currentTime, setCurrentTime] = useState(new Date())
  const [activeFilter, setActiveFilter] = useState('all')

  useEffect(() => {
    const timer = setInterval(() => setCurrentTime(new Date()), 1000)
    return () => clearInterval(timer)
  }, [])

  const infraAgents = agents.filter(a => a.layer === 'infra')
  const serviceAgents = agents.filter(a => a.layer === 'service')
  const filtered = activeFilter === 'all' ? agents
    : activeFilter === 'infra' ? infraAgents : serviceAgents

  return (
    <div className="min-h-screen bg-gray-950 text-white" dir="rtl">
      <Header time={currentTime} />

      <main className="max-w-7xl mx-auto px-6 py-8">

        {/* Stats Row */}
        <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-10">
          <StatsCard
            label="إجمالي الوكلاء"
            value={agents.length}
            icon="🤖"
            color="blue"
          />
          <StatsCard
            label="وكلاء البنية التحتية"
            value={infraAgents.length}
            icon="⚙️"
            color="purple"
          />
          <StatsCard
            label="وكلاء الخدمة"
            value={serviceAgents.length}
            icon="🛠️"
            color="green"
          />
          <StatsCard
            label="معدل النشاط"
            value="100%"
            icon="✅"
            color="emerald"
          />
        </div>

        {/* Filter Tabs */}
        <div className="flex gap-3 mb-6">
          {['all', 'infra', 'service'].map(f => (
            <button
              key={f}
              onClick={() => setActiveFilter(f)}
              className={`px-4 py-2 rounded-lg text-sm font-medium transition-all ${
                activeFilter === f
                  ? 'bg-blue-600 text-white'
                  : 'bg-gray-800 text-gray-400 hover:bg-gray-700'
              }`}
            >
              {f === 'all' ? 'الكل' : f === 'infra' ? 'البنية التحتية' : 'الخدمات'}
            </button>
          ))}
        </div>

        {/* Agents Grid */}
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
          {filtered.map(agent => (
            <AgentStatus key={agent.key} agent={agent} />
          ))}
        </div>

        {/* System Info */}
        <div className="mt-10 bg-gray-900 border border-gray-800 rounded-xl p-6">
          <h3 className="text-lg font-bold text-gray-200 mb-4">🖥️ معلومات النظام</h3>
          <div className="grid grid-cols-2 md:grid-cols-4 gap-4 text-sm">
            <div>
              <p className="text-gray-500">المنصة</p>
              <p className="text-white font-medium">Mr Code AI</p>
            </div>
            <div>
              <p className="text-gray-500">الإطار</p>
              <p className="text-white font-medium">React + Vite</p>
            </div>
            <div>
              <p className="text-gray-500">قاعدة البيانات</p>
              <p className="text-white font-medium">PostgreSQL</p>
            </div>
            <div>
              <p className="text-gray-500">نموذج الذكاء</p>
              <p className="text-white font-medium">Claude Sonnet</p>
            </div>
          </div>
        </div>

      </main>
    </div>
  )
}

### src/components/StatsCard.jsx

import React from 'react'

const colorMap = {
  blue: 'bg-blue-900/30 border-blue-800 text-blue-400',
  purple: 'bg-purple-900/30 border-purple-800 text-purple-400',
  green: 'bg-green-900/30 border-green-800 text-green-400',
  emerald: 'bg-emerald-900/30 border-emerald-800 text-emerald-400',
}

export default function StatsCard({ label, value, icon, color }) {
  return (
    <div className={`border rounded-xl p-4 ${colorMap[color]}`}>
      <div className="text-2xl mb-1">{icon}</div>
      <div className="text-2xl font-bold text-white">{value}</div>
      <div className="text-sm opacity-75">{label}</div>
    </div>
  )
}

## 📊 ما يختبره هذا المشروع

- ✅ Pipeline الكامل: تخطيط → توليد → مراجعة → حفظ → تشغيل
- ✅ React + Vite + Tailwind — مطابق للمنصة
- ✅ RTL العربية — دعم كامل
- ✅ عرض حالة الوكلاء — يعكس الوكلاء الحقيقيين
- ✅ لا axios، لا framer-motion، لا radix-ui — ملتزم بالقواعد
